	total number of business entries	新注册企业数
	total number of business exits	注销企业数
	number of businesses with turnover of zero to less than $50k	营业额 0 至 <50k 的企业数
	number of businesses with turnover of $50k to less than $200k	营业额 50k 至 <200k 的企业数
	number of businesses with turnover of $200k to less than $2m	营业额 200k 至 <2m 的企业数
	number of businesses with turnover of $2m to less than $5m	营业额 2m 至 <5m 的企业数

	number of businesses with turnover of $10m or more	营业额 ≥10m 的企业数


CREATE TABLE IF NOT EXISTS facts_businesses (
  id              BIGSERIAL PRIMARY KEY,
  region_code     varchar(20),          -- 如 State/LGA/SAx/Postcode 等
  period          NUMERIC,          
  Total_number_of_businesses numeric,
  Total_number_of_business_entries numeric,
  Total_number_of_business_exits numeric,
  Number_of_businesses_with_turnover_0_50k numeric,
  Number_of_businesses_with_turnover_50k_200k numeric,
  Number_of_businesses_with_turnover_200k_2m numeric,
  Number_of_businesses_with_turnover_2m_5m numeric,
  Number_of_businesses_with_turnover_5m_10m numeric,
  Number_of_businesses_with_turnover_10m_more numeric
);
CREATE INDEX ON facts_businesses (region_code);
CREATE INDEX ON facts_businesses (period);
